# 优化求解器对比指南

## 求解器选择决策树

```
优化问题类型
├── 线性规划（目标函数和约束均为线性）
│   ├── PuLP → 接口友好，支持多种后端求解器
│   └── scipy.optimize.linprog → 轻量，适合小规模问题
├── 整数/混合整数规划（部分变量为整数）
│   ├── PuLP + CBC → 开源免费，竞赛首选
│   └── OR-Tools → Google 出品，大规模问题表现好
├── 非线性规划（目标函数或约束非线性）
│   ├── scipy.optimize.minimize → 通用求解器
│   └── CVXPY → 凸优化专用，声明式建模
└── 多目标优化（多个冲突目标）
    ├── NSGA-II → 遗传算法，找 Pareto 前沿
    └── pymoo → Python 多目标优化框架
```

## 线性规划（LP）

### 标准形式

$$\min c^T x \quad \text{s.t.} \quad Ax \leq b, \quad x \geq 0$$

### PuLP 求解模板

```python
from pulp import *

# 1. 创建问题
prob = LpProblem("Production_Optimization", LpMaximize)

# 2. 定义变量
x1 = LpVariable("Product_A", lowBound=0, cat='Continuous')
x2 = LpVariable("Product_B", lowBound=0, cat='Continuous')

# 3. 目标函数
prob += 40 * x1 + 30 * x2, "Total_Profit"

# 4. 约束条件
prob += 2 * x1 + x2 <= 100, "Raw_Material"
prob += x1 + 3 * x2 <= 90, "Labor_Hours"
prob += x1 + x2 <= 60, "Storage"

# 5. 求解
prob.solve(PULP_CBC_CMD(msg=0))

# 6. 输出结果
print(f"状态: {LpStatus[prob.status]}")
print(f"产品A: {x1.varValue:.2f}")
print(f"产品B: {x2.varValue:.2f}")
print(f"最大利润: {value(prob.objective):.2f}")
```

### scipy.optimize.linprog 模板

```python
from scipy.optimize import linprog

# 标准形式: min c^T x, s.t. A_ub x <= b_ub, A_eq x = b_eq
c = [-40, -30]  # 注意：linprog 默认求最小值，取负号转最大化
A_ub = [[2, 1], [1, 3], [1, 1]]
b_ub = [100, 90, 60]
bounds = [(0, None), (0, None)]

result = linprog(c, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method='highs')
print(f"最优值: {-result.fun:.2f}")  # 取负号还原
print(f"最优解: x1={result.x[0]:.2f}, x2={result.x[1]:.2f}")
```

## 整数规划（IP / MILP）

### 标准形式

$$\min c^T x \quad \text{s.t.} \quad Ax \leq b, \quad x_i \in \mathbb{Z} \text{（部分或全部）}$$

### PuLP 整数规划模板

```python
from pulp import *

prob = LpProblem("Facility_Location", LpMinimize)

# 决策变量：0-1 整数变量
n_facilities = 5
n_customers = 10

# y[i] = 1 表示建设设施 i
y = [LpVariable(f"y_{i}", cat='Binary') for i in range(n_facilities)]

# x[i][j] = 1 表示客户 j 由设施 i 服务
x = [[LpVariable(f"x_{i}_{j}", cat='Binary')
      for j in range(n_customers)] for i in range(n_facilities)]

# 建设成本和服务成本
build_cost = [100, 150, 120, 80, 200]
service_cost = [[random.randint(10, 50) for _ in range(n_customers)]
                for _ in range(n_facilities)]

# 目标函数：最小化总成本
prob += (lpSum(build_cost[i] * y[i] for i in range(n_facilities)) +
         lpSum(service_cost[i][j] * x[i][j]
                for i in range(n_facilities)
                for j in range(n_customers)))

# 约束：每个客户恰好被一个设施服务
for j in range(n_customers):
    prob += lpSum(x[i][j] for i in range(n_facilities)) == 1

# 约束：只有建设的设施才能服务客户
for i in range(n_facilities):
    for j in range(n_customers):
        prob += x[i][j] <= y[i]

prob.solve(PULP_CBC_CMD(msg=0))
print(f"总成本: {value(prob.objective):.0f}")
```

### OR-Tools 整数规划模板

```python
from ortools.linear_solver import pywraplp

solver = pywraplp.Solver.CreateSolver('CBC')

# 定义变量
x = {}
for i in range(5):
    for j in range(10):
        x[i, j] = solver.IntVar(0, 1, f'x_{i}_{j}')

y = [solver.IntVar(0, 1, f'y_{i}') for i in range(5)]

# 目标函数
objective = solver.Objective()
for i in range(5):
    objective.SetCoefficient(y[i], build_cost[i])
    for j in range(10):
        objective.SetCoefficient(x[i, j], service_cost[i][j])
objective.SetMinimization()

# 约束
for j in range(10):
    ct = solver.Constraint(1, 1)
    for i in range(5):
        ct.SetCoefficient(x[i, j], 1)

solver.Solve()
print(f"最优值: {objective.Value():.0f}")
```

## 非线性规划（NLP）

### 标准形式

$$\min f(x) \quad \text{s.t.} \quad g_i(x) \leq 0, \quad h_j(x) = 0$$

### scipy.optimize.minimize 模板

```python
from scipy.optimize import minimize, LinearConstraint, NonlinearConstraint

# 目标函数
def objective(x):
    return x[0]**2 + x[1]**2 + x[0]*x[1] - 4*x[0] - 5*x[1]

# 非线性约束
def constraint1(x):
    return x[0] + x[1] - 3  # >= 0

def constraint2(x):
    return x[0]**2 + x[1] - 5  # >= 0

constraints = [
    {'type': 'ineq', 'fun': constraint1},
    {'type': 'ineq', 'fun': constraint2}
]

bounds = [(0, None), (0, None)]
x0 = [1.0, 1.0]  # 初始猜测

result = minimize(objective, x0, method='SLSQP',
                  bounds=bounds, constraints=constraints)
print(f"最优解: {result.x}")
print(f"最优值: {result.fun:.4f}")
print(f"是否收敛: {result.success}")
```

### CVXPY 凸优化模板

```python
import cvxpy as cp
import numpy as np

# 定义变量
x = cp.Variable(2)

# 目标函数（必须是凸的）
objective = cp.Minimize(cp.sum_squares(x) - 4*x[0] - 5*x[1])

# 约束
constraints = [
    x[0] + x[1] <= 10,
    x >= 0,
    x[0]**2 + x[1] <= 20  # 凸约束
]

prob = cp.Problem(objective, constraints)
prob.solve()

print(f"最优值: {prob.value:.4f}")
print(f"最优解: {x.value}")
```

## 多目标优化

### Pareto 前沿概念

当存在多个冲突目标时，不存在单一最优解，而是一组 Pareto 最优解：
- 解 A 支配解 B：A 在所有目标上不差于 B，且至少在一个目标上严格优于 B
- Pareto 最优解：不被任何其他解支配的解集合

### pymoo NSGA-II 模板

```python
from pymoo.algorithms.moo.nsga2 import NSGA2
from pymoo.problems import get_problem
from pymoo.optimize import minimize
from pymoo.visualization.scatter import Scatter

# 定义问题（以 ZDT1 为例）
problem = get_problem("zdt1")

# 创建算法
algorithm = NSGA2(pop_size=100)

# 求解
res = minimize(problem,
               algorithm,
               ('n_gen', 200),
               seed=1,
               verbose=False)

# 可视化 Pareto 前沿
plot = Scatter(title="Pareto Front")
plot.add(res.F)
plot.show()
```

### 自定义多目标问题

```python
from pymoo.core.problem import ElementwiseProblem
from pymoo.algorithms.moo.nsga2 import NSGA2
from pymoo.optimize import minimize

class MyProblem(ElementwiseProblem):
    def __init__(self):
        super().__init__(
            n_var=2, n_obj=2, n_constr=0,
            xl=[0.0, 0.0], xu=[5.0, 5.0]
        )

    def _evaluate(self, x, out, *args, **kwargs):
        f1 = x[0]**2 + x[1]**2  # 目标1：最小化距离
        f2 = (x[0]-3)**2 + (x[1]-2)**2  # 目标2：最小化到(3,2)的距离
        out["F"] = [f1, f2]

problem = MyProblem()
algorithm = NSGA2(pop_size=50)
res = minimize(problem, algorithm, ('n_gen', 200), seed=1, verbose=False)

print(f"找到 {len(res.F)} 个 Pareto 最优解")
for i, (f1, f2) in enumerate(res.F):
    print(f"  解{i}: f1={f1:.3f}, f2={f2:.3f}, x={res.X[i]}")
```

## 常见陷阱

### 局部最优

- 非凸问题中，梯度下降可能陷入局部最优
- **解决方案**：多起点搜索、模拟退火、遗传算法、粒子群优化

```python
# 多起点策略
from scipy.optimize import minimize
import numpy as np

best_result = None
best_value = np.inf

for _ in range(100):
    x0 = np.random.uniform(-10, 10, size=2)
    result = minimize(objective, x0, method='SLSQP')
    if result.success and result.fun < best_value:
        best_value = result.fun
        best_result = result

print(f"全局最优近似: {best_result.x}, 值: {best_result.fun:.4f}")
```

### 收敛失败

- 迭代次数不足、步长过大/过小、初始点选择不当
- **解决方案**：调整 `maxiter`、尝试不同求解器、改进初始点

### 变量范围设置不当

- 未设置 `bounds` 导致求解空间过大
- 整数变量忘记设为 `cat='Integer'` 或 `cat='Binary'`
- **解决方案**：根据实际物理意义设置合理的上下界

### 数值问题

- 大尺度差异导致数值不稳定（如目标函数值在 $10^6$ 级，约束在 $10^{-3}$ 级）
- **解决方案**：变量归一化、约束标准化

## 求解器对比总结

| 求解器 | 类型 | 开源 | 适用规模 | Python 包 |
|--------|------|------|---------|-----------|
| CBC | LP/MIP | 是 | 中小规模 | PuLP, OR-Tools |
| GLPK | LP/MIP | 是 | 小规模 | PuLP |
| HiGHS | LP/MIP | 是 | 中大规模 | scipy, PuLP |
| Gurobi | LP/MIP/NLP | 商用 | 大规模 | gurobipy |
| CPLEX | LP/MIP | 商用 | 大规模 | docplex |
| IPOPT | NLP | 是 | 中大规模 | cyipopt |
| SLSQP | NLP | 是 | 小规模 | scipy |
| CVXPY | 凸优化 | 是 | 中规模 | cvxpy |
